<!--INTRODUCCIÓN button and message -->

<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <title>Mi proyecto con Ext JS</title>

    <!-- CSS de Ext JS: Estilos necesarios para que los componentes se vean bien -->
    <link rel="stylesheet" type="text/css" href="ext-4/resources/css/ext-all.css">                                                                                     

    <!-- JavaScript de Ext JS: Biblioteca principal para crear componentes -->
    <script type="text/javascript" src="ext-4/ext-all.js"></script>
</head>
<body>
    <script type="text/javascript">
        // Ext.onReady se asegura de que el código dentro se ejecute
        // sólo cuando la página y Ext JS estén completamente cargados
        Ext.onReady(function() {

            // Mostrar un mensaje emergente de bienvenida para confirmar que Ext JS funciona
            Ext.Msg.alert('Ext JS', '¡Ext JS está funcionando correctamente!');

            // Crear un botón con un menú desplegable
            Ext.create('Ext.button.Button', {
                text: 'Mi primer botón',          // Texto que se muestra en el botón
                renderTo: Ext.getBody(),          // Indica que el botón se dibuje dentro del <body> del HTML
                menu: [                          // Define un menú con dos opciones dentro del botón
                    {
                        text: 'Item 1',          // Texto del primer item del menú
                        handler: function() {    // Función que se ejecuta cuando se selecciona "Item 1"
                            Ext.Msg.alert('Item', 'Has seleccionado el Item 1');
                        }
                    },
                    {
                        text: 'Item 2',          // Texto del segundo item del menú
                        handler: function() {    // Función que se ejecuta cuando se selecciona "Item 2"
                            Ext.Msg.alert('Item', 'Has seleccionado el Item 2');
                        }
                    }
                ]
            });

        });
    </script>
</body>
</html>
